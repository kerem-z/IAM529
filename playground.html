Lorenz System Mathematics

Testing LaTeX rendering with Lorenz attractor equations and symbols:

The Lorenz system: $\dot{x} = \sigma(y-x)$, $\dot{y} = x(\rho-z)-y$, $\dot{z} = xy-\beta z$

Parameters: $\sigma = 10$, $\rho = 28$, $\beta = \frac{8}{3}$

Equilibrium points: $C_0 = (0,0,0)$ and $C_{\pm} = (\pm\sqrt{\beta(\rho-1)}, \pm\sqrt{\beta(\rho-1)}, \rho-1)$

Complete system:

$$ \begin{aligned} \frac{dx}{dt} &= \sigma (y - x) \\ \frac{dy}{dt} &= x (\rho - z) - y \\ \frac{dz}{dt} &= x y - \beta z \end{aligned} $$

Lyapunov exponents: $\lambda_1 > 0$, $\lambda_2 = 0$, $\lambda_3 < 0$

Strange attractor dimension: $D_L = 2 + \frac{\lambda_1}{|\lambda_3|} \approx 2.06$

# Lorenz system numerical integration
import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import odeint

def lorenz_system(state, t, sigma, rho, beta):
    x, y, z = state
    dxdt = sigma * (y - x)
    dydt = x * (rho - z) - y
    dzdt = x * y - beta * z
    return [dxdt, dydt, dzdt]

# Parameters
sigma, rho, beta = 10.0, 28.0, 8.0/3.0
initial_state = [1.0, 1.0, 1.0]
t = np.linspace(0, 25, 10000)

# Solve system
trajectory = odeint(lorenz_system, initial_state, t, args=(sigma, rho, beta))

# Extract coordinates
x, y, z = trajectory.T

# 3D plot
fig = plt.figure(figsize=(12, 9))
ax = fig.add_subplot(111, projection='3d')
ax.plot(x, y, z, lw=0.5, color='blue')
ax.set_xlabel('X')
ax.set_ylabel('Y')
ax.set_zlabel('Z')
ax.set_title('Lorenz Attractor')
plt.show()

# Lorenz system in Julia
using DifferentialEquations, Plots

function lorenz!(du, u, p, t)
    σ, ρ, β = p
    du[1] = σ * (u[2] - u[1])
    du[2] = u[1] * (ρ - u[3]) - u[2]
    du[3] = u[1] * u[2] - β * u[3]
end

# Parameters
σ, ρ, β = 10.0, 28.0, 8.0/3.0
u0 = [1.0, 1.0, 1.0]
tspan = (0.0, 25.0)
p = [σ, ρ, β]

# Solve ODE
prob = ODEProblem(lorenz!, u0, tspan, p)
sol = solve(prob, Tsit5(), reltol=1e-8, abstol=1e-8)

# Extract solution

     xlabel="X",

     title="Lorenz Attractor")

// Lorenz system in JavaScript
function lorenzSystem(x, y, z, sigma, rho, beta, dt) {
    const dx = sigma * (y - x) * dt;
    const dy = (x * (rho - z) - y) * dt;
    const dz = (x * y - beta * z) * dt;
    return [dx, dy, dz];
}

function integrateLorenz(steps = 10000) {
    const sigma = 10.0, rho = 28.0, beta = 8.0/3.0;
    const dt = 0.01;
    
    let x = 1.0, y = 1.0, z = 1.0;
    const trajectory = [];
    
    for (let i = 0; i < steps; i++) {
        trajectory.push([x, y, z]);
        
        const [dx, dy, dz] = lorenzSystem(x, y, z, sigma, rho, beta, dt);
        x += dx;
        y += dy;
        z += dz;
    }
    
    return trajectory;
}

const points = integrateLorenz();

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

typedef struct {
    double x, y, z;
} Point3D;

void lorenz_derivative(Point3D state, Point3D *derivative, 
                      double sigma, double rho, double beta) {
    derivative->x = sigma * (state.y - state.x);
    derivative->y = state.x * (rho - state.z) - state.y;
    derivative->z = state.x * state.y - beta * state.z;
}

void runge_kutta_4(Point3D *state, double dt, 
                   double sigma, double rho, double beta) {
    Point3D k1, k2, k3, k4, temp;
    
    // k1
    lorenz_derivative(*state, &k1, sigma, rho, beta);
    
    // k2
    temp.x = state->x + 0.5 * dt * k1.x;
    temp.y = state->y + 0.5 * dt * k1.y;
    temp.z = state->z + 0.5 * dt * k1.z;
    lorenz_derivative(temp, &k2, sigma, rho, beta);
    
    // k3
    temp.x = state->x + 0.5 * dt * k2.x;
    temp.y = state->y + 0.5 * dt * k2.y;
    temp.z = state->z + 0.5 * dt * k2.z;
    lorenz_derivative(temp, &k3, sigma, rho, beta);
    
    // k4
    temp.x = state->x + dt * k3.x;
    temp.y = state->y + dt * k3.y;
    temp.z = state->z + dt * k3.z;
    lorenz_derivative(temp, &k4, sigma, rho, beta);
    
    // Update state
    state->x += (dt/6.0) * (k1.x + 2*k2.x + 2*k3.x + k4.x);
    state->y += (dt/6.0) * (k1.y + 2*k2.y + 2*k3.y + k4.y);
    state->z += (dt/6.0) * (k1.z + 2*k2.z + 2*k3.z + k4.z);
}

int main() {
    const double sigma = 10.0, rho = 28.0, beta = 8.0/3.0;
    const double dt = 0.01;
    const int steps = 100000;
    
    Point3D state = {1.0, 1.0, 1.0};
    
    printf("# Lorenz Attractor Data\n");
    printf("# X Y Z\n");
    
    for (int i = 0; i < steps; i++) {
        if (i % 100 == 0) {  // Print every 100th point
            printf("%.6f %.6f %.6f\n", state.x, state.y, state.z);
        }
        runge_kutta_4(&state, dt, sigma, rho, beta);
    }
    
    return 0;
}

function lorenz_attractor()
    % Lorenz system parameters
    sigma = 10.0;
    rho = 28.0;
    beta = 8.0/3.0;
    
    % Initial conditions
    y0 = [1.0; 1.0; 1.0];
    
    % Time span
    tspan = [0 25];
    
    % Solve ODE
    [t, y] = ode45(@(t,y) lorenz_ode(t, y, sigma, rho, beta), tspan, y0);
    
    % Extract coordinates
    x = y(:,1);
    y_coord = y(:,2);
    z = y(:,3);
    
    % 3D plot
    figure;
    plot3(x, y_coord, z, 'b-', 'LineWidth', 0.5);
    xlabel('X');
    ylabel('Y');
    zlabel('Z');
    title('Lorenz Attractor');
    grid on;
    axis equal;
    view(3);
end

function dydt = lorenz_ode(t, y, sigma, rho, beta)
    % Lorenz system differential equations
    dydt = zeros(3,1);
    dydt(1) = sigma * (y(2) - y(1));
    dydt(2) = y(1) * (rho - y(3)) - y(2);
    dydt(3) = y(1) * y(2) - beta * y(3);
end

% Run the simulation
lorenz_attractor();
